    global Database.QueryLocator start(Database.BatchableContext bc) {
        
        return Database.getQueryLocator([
            SELECT Id  , Compensation_Credits_Overall__c
            FROM Booking__c where Compensation_Credits_Overall__c = null
        ]);
    }
    
    global void execute(Database.BatchableContext bc, List<Booking__c> bookings) {
        
      
        Set<Id> bookingIds = new Map<Id, Booking__c>(bookings).keySet();
        
      
        List<Compensation_Details__c> compDetails = [
            SELECT Id, 
            Booking__c,           
            Sub_Total__c,         
            Payment_Type__c       
            FROM Compensation_Details__c
            WHERE Booking__c IN :bookingIds
            AND (
                Payment_Type__c = 'Monthly Accrual' 
                OR Payment_Type__c = 'One-time Accrual'
            )
        ];
        
        
        Map<Id, Decimal> bookingCreditMap = new Map<Id, Decimal>();
        
        for (Compensation_Details__c cd : compDetails) {
            if (!bookingCreditMap.containsKey(cd.Booking__c)) {
                bookingCreditMap.put(cd.Booking__c, 0);
            }
            Decimal subTotal = cd.Sub_Total__c != null ? cd.Sub_Total__c : 0;
            bookingCreditMap.put(
                cd.Booking__c, 
                bookingCreditMap.get(cd.Booking__c) + subTotal
            );
        }
        
        
        List<Booking__c> bookingsToUpdate = new List<Booking__c>();
        
        for (Booking__c b : bookings) {
            Decimal totalCredits = bookingCreditMap.containsKey(b.Id) 
                ? bookingCreditMap.get(b.Id) 
                : 0;
            bookingsToUpdate.add(new Booking__c(
                Id = b.Id,
                Compensation_Credits_Overall__c = totalCredits 
            ));
        }
        
        if (!bookingsToUpdate.isEmpty()) {
            update bookingsToUpdate;
        }
    }
